\newcommand*{\pRC}[2]{%
  \begin{scope}[xshift=#1, yshift=#2] 
    \draw (0,0) -- (0.3, 0) -- (0.3, 0.5) to[R, ] (1.7, 0.5) -- (1.7, 0) -- (2,0); 
    \draw (0.3, 0) to[, *-,] (0.3, -0.5) to [C, ] (1.7,-0.5) -- (1.7, 0); 
  \end{scope}
}
