\pgfmathsetmacro{\zz}{0.99}

\begin{tikzpicture}
\begin{axis}[
clip=false,

height=4cm,
axis lines=left,
axis line style={->},
%xtick=\empty,
%ytick=\empty,
%xmin=-1,

]
\addplot[solid,thick,domain=0:10, samples=400] {1 - exp(-\ww*\zz*x) / sqrt(1-\zz * \zz) * sin(deg(\ww*sqrt(1 - \zz * \zz)*x) + acos(\zz))};
\end{axis}
\end{tikzpicture}
